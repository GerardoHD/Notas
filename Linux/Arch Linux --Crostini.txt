#!/bin/bash

ctrl-alt-t

vmc start termina

lxc stop penguin

lxc launch images:archlinux arch

lxc exec arch -- bash

#
ip -4 a show dev eth0
# ^ See if there's a network connection.
#   non-empty output = good (skip)
#   empty output = proceed below
dhcpcd eth0
ping -c 3 archlinux.org
sudo pacman -Syu dhclient
sudo systemctl enable --now dhclient@eth0
#

pacman -Syyu --noconfirm base-devel dialog wget git sudo vim

curl -L https://git.io/JzpKF | bash

#Distros-Setup / Etapa 1

sudo poweroff

==========================

lxc start arch

lxc console arch

#Distros-Setup / Etapa 2

sudo poweroff

==========================

lxc start arch

lxc console arch

#Distros-Setup / Etapa 3

sudo poweroff

exit

vmc stop termina

==========================

Abrir terminal, si todo funciona correctamente cambiar contenedores:

lxc stop arch
lxc stop penguin
lxc rename penguin google
lxc rename arch penguin
exit
vmc list
vmc stop termina
