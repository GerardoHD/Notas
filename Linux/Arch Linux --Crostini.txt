#!/bin/bash

sudo poweroff

ctrl-alt-t

vsh termina

df -hT

lxc list

lxc stop penguin

lxc delete penguin

lxc image list

lxc image delete $FINGERPRINT

df -hT

exit

vmc stop termina

vmc start termina

lxc launch images:archlinux penguin

lxc config set penguin security.nesting true

lxc exec penguin -- bash

#
ip -4 a show dev eth0
# ^ See if there's a network connection.
#   non-empty output = good (skip)
#   empty output = proceed below
dhcpcd eth0
ping -c 3 archlinux.org
sudo pacman -Syu --noconfirm dhclient
sudo systemctl enable --now dhclient@eth0
#

pacman -Syyu --noconfirm base-devel dialog wget git sudo vim

curl -L https://git.io/JrQil | bash

#Distros-Setup / Etapa 1

sudo poweroff

==========================

lxc start penguin

lxc console penguin

#Distros-Setup / Etapa 2

sudo poweroff

==========================

lxc start penguin

lxc console penguin

#Distros-Setup / Etapa 3

sudo poweroff

exit

vmc stop termina
